// Efficient Hash Table Implementation
class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        unordered_map<int, int> table;

        for(int i = 0; i < nums.size(); i++){
            if(table.contains(nums[i]) && abs(table[nums[i]] - i) <= k){
                return true;
            }
            table[nums[i]] = i;
        }
        return false;
    }
};

// Sliding Window Solution

class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        unordered_map<int, int> numIndexMap;  // Store the most recent index of each number

        for (int i = 0; i < nums.size(); i++) {
            // Check if the number is already in the map
            if (numIndexMap.find(nums[i]) != numIndexMap.end()) {
                // Check the condition if the difference of indices is less than or equal to k
                if (i - numIndexMap[nums[i]] <= k) {
                    return true;  // Found a nearby duplicate
                }
            }

            // Update the map with the current index of the number
            numIndexMap[nums[i]] = i;
        }

        return false;  // No nearby duplicates found
    }
};
